💡 EXEMPLE D'UTILISATION
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

// 1. Activer "Accès au contexte Budibase" dans les propriétés du plugin
// 2. Utiliser les APIs Budibase directement:

// Rechercher des utilisateurs actifs
budibase.API.searchTable({
  tableId: "ta_users",
  query: {
    equal: { status: "active" }
  },
  limit: 50
}).then(result => {
  budibase.notify.success(`${result.data.length} utilisateurs trouvés`)
  console.log("Résultats:", result.data)
})

// Créer un nouveau produit
budibase.API.saveRow({
  tableId: "ta_products",
  name: "Nouveau produit",
  price: 99.99,
  status: "active"
}).then(result => {
  budibase.notify.success("Produit créé avec succès!")
})
